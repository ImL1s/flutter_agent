String buildPrompt(WidgetDescriptor uiState, String task) {
  if (promptTemplate != null) {
    return promptTemplate!.format(
      uiState: uiState,
      task: task,
      actionNames: actionRegistry.registeredActions,
    );
  }
  return const DefaultPromptTemplate().format(
    uiState: uiState,
    task: task,
    actionNames: actionRegistry.registeredActions,
  );
}
